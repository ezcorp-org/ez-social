**Platform detection (`src/lib/utils/platform.ts`):**

EXTEND the existing file (do NOT replace existing `platformStyles` export). Add a `detectPlatform(url)` function:

```typescript
const PLATFORM_URL_PATTERNS: Record<string, RegExp[]> = {
  twitter: [/(?:twitter\.com|x\.com)/i],
  linkedin: [/linkedin\.com/i],
  reddit: [/(?:reddit\.com|old\.reddit\.com)/i],
  blog: [/(?:medium\.com|substack\.com|dev\.to|hashnode\.dev)/i],
};

export function detectPlatform(url: string): string | null {
  try {
    const hostname = new URL(url).hostname;
    for (const [platform, patterns] of Object.entries(PLATFORM_URL_PATTERNS)) {
      if (patterns.some(p => p.test(hostname))) return platform;
    }
  } catch {
    // Invalid URL — return null
  }
  return null;
}
```

**Scraper client (`src/lib/server/services/scraper.ts`):**

Create a lightweight fetch-based scraper client. The scraper service is external (Playwright on a VPS). The client must handle: (1) missing SCRAPER_SERVICE_URL env var, (2) service timeout, (3) service errors — all gracefully returning null.

```typescript
interface ScrapeResult {
  content: string;
  author?: string;
}

export async function scrapeUrl(scraperBaseUrl: string | undefined, targetUrl: string): Promise<ScrapeResult | null> {
  if (!scraperBaseUrl) return null; // No scraper configured

  try {
    const res = await fetch(`${scraperBaseUrl}/scrape`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ url: targetUrl }),
      signal: AbortSignal.timeout(15000), // 15s timeout
    });
    if (!res.ok) return null;
    return await res.json() as ScrapeResult;
  } catch {
    return null; // Network error, timeout, etc.
  }
}
```
